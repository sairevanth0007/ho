/**
 * @file routes/auth.routes.js
 * @description Authentication routes for OAuth login, logout, and user profile.
 * @author GIDE
 */

import { Router } from 'express';
import passport from 'passport'; // Import configured passport from passport-setup
import { authController } from '../controllers/auth.controller.js';
import { isAuthenticated } from '../middlewares/auth.middleware.js';
import { OAuthProviders } from '../constants/index.js'; 
import { bridge } from '../bridge.js';

const router = Router();

// --- OAuth Initiation Routes ---
// These routes will redirect to the respective OAuth provider.
// We add a middleware to capture referralCode from query and store in session.

const initiateOAuth = (provider, scope) => (req, res, next) => {
    const { referralCode } = req.query;
    if (referralCode) {
        req.session.referralCode = referralCode.toString().trim().toUpperCase();
    }
    passport.authenticate(provider, { scope })(req, res, next);
};

/**
 * @route GET /api/v1/auth/google
 * @description Initiates Google OAuth authentication.
 * @query {string} [referralCode] - Optional referral code.
 * @group Auth - Authentication operations
 * @returns {void} Redirects to Google's OAuth consent screen.
 */
router.get('/google', initiateOAuth(OAuthProviders.GOOGLE, ['profile', 'email']));

/**
 * @route GET /api/v1/auth/microsoft
 * @description Initiates Microsoft OAuth authentication.
 * @query {string} [referralCode] - Optional referral code.
 * @group Auth - Authentication operations
 * @returns {void} Redirects to Microsoft's OAuth consent screen.
 */
router.get('/microsoft', initiateOAuth(OAuthProviders.MICROSOFT, ['openid', 'profile', 'email', 'User.Read']));

/**
 * @route GET /api/v1/auth/github
 * @description Initiates GitHub OAuth authentication.
 * @query {string} [referralCode] - Optional referral code.
 * @group Auth - Authentication operations
 * @returns {void} Redirects to GitHub's OAuth consent screen.
 */
router.get('/github', initiateOAuth(OAuthProviders.GITHUB, ['user:email', 'read:user']));


// --- OAuth Callback Routes ---
// Providers will redirect back to these URLs after user grants/denies permission.
// Passport will handle the code exchange and call the verify callback.

const oauthCallbackOptions = {
    failureRedirect: `${bridge.BASE_URL}/api/v1/auth/oauth/failure`, // Redirect to our failure handler
    // successRedirect: Can be set, but custom handler gives more flexibility e.g. for JWTs or different redirects
    // session: true is default and needed for session-based auth
};

/**
 * @route GET /api/v1/auth/google/callback
 * @description Google OAuth callback URL.
 * @group Auth - Authentication operations
 * @returns {void} Redirects to frontend on success or failure.
 */
router.get('/google/callback',
    passport.authenticate(OAuthProviders.GOOGLE, oauthCallbackOptions),
    authController.handleOAuthSuccess // Called on successful authentication
);

/**
 * @route GET /api/v1/auth/microsoft/callback
 * @description Microsoft OAuth callback URL.
 * @group Auth - Authentication operations
 * @returns {void} Redirects to frontend on success or failure.
 */
router.get('/microsoft/callback',
    passport.authenticate(OAuthProviders.MICROSOFT, oauthCallbackOptions),
    authController.handleOAuthSuccess
);

/**
 * @route GET /api/v1/auth/github/callback
 * @description GitHub OAuth callback URL.
 * @group Auth - Authentication operations
 * @returns {void} Redirects to frontend on success or failure.
 */
router.get('/github/callback',
    passport.authenticate(OAuthProviders.GITHUB, oauthCallbackOptions),
    authController.handleOAuthSuccess
);

// --- OAuth Failure Route ---
/**
 * @route GET /api/v1/auth/oauth/failure
 * @description Generic OAuth failure handler.
 * @group Auth - Authentication operations
 * @returns {void} Redirects to frontend login page with an error message.
 */
router.get('/oauth/failure', authController.handleOAuthFailure);


// --- Other Authentication Routes ---

/**
 * @route POST /api/v1/auth/logout
 * @description Logs out the current user.
 * @group Auth - Authentication operations
 * @security BasicAuth
 * @returns {ApiResponse.model} 200 - Logout successful
 * @returns {ApiError.model} 500 - Logout failed
 */
router.post('/logout', isAuthenticated, authController.logoutUser);

/**
 * @route GET /api/v1/auth/me
 * @description Gets the profile of the currently authenticated user.
 * @group Auth - Authentication operations
 * @security BasicAuth
 * @returns {ApiResponse.model} 200 - User profile fetched successfully
 * @returns {ApiError.model} 401 - Unauthorized
 */
router.get('/me', isAuthenticated, authController.getCurrentUser);

export default router;
